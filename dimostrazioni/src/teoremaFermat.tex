\begin{Definizioni necessarie} %Questo vorrebbe essere un abstract prima del testo effettivo
    \begin{itemize}
        \item \(x_0\) è un punto stazionario se \(\testit{f}(x_0) = 0\)
        \item \(x_0\) è un punto di ottimo se è un punto di massimo o di minimo locale
        \item \(x_M\) è un punto di massimo locale se  \( M = \textit{f}(x_M) \geqslant \textit{f}(x) \forall  x \in A\) dove M è il valore massimo locale
        \item \(x_M\) è un punto di minimo locale se  \( m = \textit{f}(x_m) \leqslant \textit{f}(x) \forall  x \in A\) dove m è il valore minimo locale
    \end{itemize}
\end{Definizioni necessarie}
