    \chapter{Teorema di Fermat}

        \section*{Enunciato}

        \subsection*{Ipotesi}

            Sia \(f(x)\) una funzione tale che
            \begin{align*}
                f : A = [a, b] &\longrightarrow \mathbb{R}\\
                x &\longmapsto y = f(x) 
            \end{align*}

            Supponiamo inoltre che:

            \begin{enumerate}
                \indentitem \item \(x_0 \in A\);
                \indentitem \item \(f\) è derivabile in \(A\);
                \indentitem \item \(x_0\) è un punto di ottimo.
            \end{enumerate}

        \subsection*{Tesi}

            \[f'(x) = 0\]

        \section*{Dimostrazione}

            \medskip

            % Nota: sono attaccate per evitare multiple righe vuote
            Dimostriamo l'enunciato per \(n = 0\):
            \begin{align*}
                (1 + x)^0 \, &\geqslant \, 1 + 0 x\\
                        1 \, &\geqslant \, 1
            \end{align*}

            Possiamo perciò considerare l'enunciato vero al passo \(n\).

            \medskip

            % Nota: si veda sopra
            Dimostriamolo per \(n + 1\):
            \begin{align*}
                (1 + x)^{n + 1} \, =& \, (1 + x)(1 + x)^n\\
                                \geqslant& \, (1 + x)(1 + nx) & \text{\emph{Per ipotesi induttiva}}\\
                                =& \, 1 + nx + x + nx^2\\
                                =& \, 1 + x (n + 1) + nx^2\\
                                \geqslant& \, 1 + x (n + 1) & \text{\emph{Per l'enunciato del teorema}}
            \end{align*}

            Abbiamo quindi dimostrato la disuguaglianza di Bernoulli.
